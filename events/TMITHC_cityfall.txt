add_namespace = cityfall

# France captures Germania
news_event = {
	id = cityfall.1
	title = cityfall.1.t
	desc = cityfall.1.d
	picture = GFX_news_event_france_germania
	
	major = yes
	
	trigger = {
		361 = { is_controlled_by = FRA }
		361 = { is_owned_by = GER }
		FRA = { has_war_with = GER }
		NOT = { has_global_flag = fall_of_germania }
	}
	
	immediate = {
		set_global_flag = fall_of_germania
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = cityfall.1.a
		trigger = {
			NOT = {
				TAG = FRA
				TAG = GER
			}
		}
	}
	option = {
		name = cityfall.1.b
		trigger = { TAG = FRA }
	}
	option = {
		name = cityfall.1.c
		trigger = { TAG = GER }
	}
}
